//count
select pemenang Pemenang, count(*) Jumlah from 
(SELECT * FROM pkl.bantul b UNION ALL 
 SELECT * FROM pkl.gunungkidul gk UNION ALL 
 SELECT * FROM pkl.jogjakota jk UNION ALL 
 SELECT * FROM pkl.jogjaprov jp UNION ALL 
 SELECT * FROM pkl.kulonprogo kp UNION ALL 
 SELECT * FROM pkl.sleman s UNION ALL 
 SELECT * FROM pkl.ugm u) x 
 group by pemenang order by count(*) desc limit 10
